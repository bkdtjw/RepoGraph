 797	def _rg_focus_peek(sess, cur_turn, want_labels=None):
 798	    """取最近**未过期 (TTL)** 且类型相容的焦点条目; 无则 None。want_labels=None 表任意 label。"""
 799	    for e in reversed(sess.get("rg_focus") or []):
 800	        if not isinstance(e, dict):
 801	            continue
 802	        try:
 803	            age = cur_turn - int(e.get("turn", -(10 ** 9)))
 804	        except (TypeError, ValueError):
 805	            continue
 806	        if age > _RG_FOCUS_TTL:
 807	            continue
 808	        if want_labels and e.get("label") not in want_labels:
 809	            continue
 810	        if e.get("entity_id"):

 812	    return None
