  77	def build_repo_card(store: GraphStore, summary: Optional[str] = None) -> dict:
  78	    """装配 level-0 卡片的**确定性**字段（+ 可选 summary）。纯函数、零网络。
  79	
  80	    与 ``context.build_overview`` 口径一致：顶层模块按 ``loc`` 降序、热点函数按 MODIFIES
  81	    入度、核心概念按 IMPLEMENTS 落点数；``entrypoints`` 聚合 ``is_endpoint`` 为真的函数
  82	    （本图为 0，则如实产空）。返回可 JSON 序列化的卡片 dict。
  83	    """
  84	    counts = store.counts().get("nodes", {})
  85	    stats = {
  86	        "modules": counts.get("Module", 0),
  87	        "classes": counts.get("Class", 0),
  88	        "functions": counts.get("Function", 0),
  89	        "commits": counts.get("Commit", 0),
  90	        "concepts": counts.get("Concept", 0),
  91	    }
  92	
  93	    # 顶层模块：按 loc 降序 top N

 148	def render_card_text(card: dict) -> str:
 149	    """把卡片渲染为中文注入文本（含全部确定性事实 + 可选 summary）。
 150	
 151	    **恒含规模事实行**（模块/类/函数/提交/概念计数），使 meta 路由无论是否有 summary
 152	    都满足 L0 事实达标（≥3），与 ``build_overview`` 的事实口径一致。
 153	    """
 154	    s = card.get("stats", {})
 155	    lines: list[str] = []
 156	    lines.append("【仓库卡片】（来源: 图谱统计 · level-0 repo_card）")
 157	    lines.append(f"仓库: {card.get('repo', '(unknown)')}")
 158	    lines.append(
 159	        f"规模: 模块 {s.get('modules', 0)} · 类 {s.get('classes', 0)} · "
 160	        f"函数 {s.get('functions', 0)} · 提交 {s.get('commits', 0)} · "
 161	        f"概念 {s.get('concepts', 0)}"
 162	    )
 163	
 164	    summary = card.get("summary")
 165	    if summary:
 166	        lines.append("")
 167	        lines.append("【概述】（来源: 索引期网关生成 · 已过专名白名单校验）")
 168	        lines.append(summary)
 169	
 170	    mods = card.get("top_modules") or []
 171	    if mods:
 172	        lines.append("")
 173	        lines.append("【顶层模块】（来源: Module 按 loc 降序）")
 174	        for m in mods:
 175	            lines.append(f"- {m.get('path', '?')} (loc={m.get('loc', '?')}): "
 176	                         f"{m.get('doc') or '（无 docstring）'}")
 177	
 178	    hot = card.get("hot_functions") or []
 179	    if hot:
 180	        lines.append("")
 181	        lines.append("【热点函数】（来源: MODIFIES 计数，改动最频繁）")
 182	        for h in hot:
 183	            lines.append(f"- {h.get('qualname', '')} (被 {h.get('modifies', 0)} 次提交修改)")
 184	
 185	    core = card.get("core_concepts") or []
 186	    if core:
 187	        lines.append("")
 188	        lines.append("【核心概念】（来源: IMPLEMENTS 落点最多）")
 189	        for c in core:
 190	            lines.append(f"- {c.get('name', '')} (落 {c.get('impl', 0)} 处实现): "
 191	                         f"{c.get('desc') or ''}")
 192	
 193	    eps = card.get("entrypoints") or []
 194	    lines.append("")
 195	    if eps:
 196	        lines.append("【入口点】（来源: Function.is_endpoint 聚合）")
 197	        for e in eps:
 198	            hm = e.get("http_method") or "?"
 199	            rp = e.get("route_path") or "?"
 200	            lines.append(f"- {e.get('qualname', '')} [{hm} {rp}]")
 201	    else:
 202	        lines.append("【入口点】（来源: Function.is_endpoint 聚合）: （本图无 HTTP 端点）")
 203	
 204	    return "\n".join(lines) + "\n"
