  78	def build_repo_card(store: GraphStore, summary: Optional[str] = None) -> dict:

 149	def render_card_text(card: dict) -> str:
 150	    """把卡片渲染为中文注入文本（含全部确定性事实 + 可选 summary）。
 151	
 152	    **恒含规模事实行**（模块/类/函数/提交/概念计数），使 meta 路由无论是否有 summary
 153	    都满足 L0 事实达标（≥3），与 ``build_overview`` 的事实口径一致。
 154	    """

 156	    lines: list[str] = []
 157	    lines.append("【仓库卡片】（来源: 图谱统计 · level-0 repo_card）")
 158	    lines.append(f"仓库: {card.get('repo', '(unknown)')}")
 159	    lines.append(
 160	        f"规模: 模块 {s.get('modules', 0)} · 类 {s.get('classes', 0)} · "
 161	        f"函数 {s.get('functions', 0)} · 提交 {s.get('commits', 0)} · "
 162	        f"概念 {s.get('concepts', 0)}"
 163	    )
 164	
 165	    summary = card.get("summary")
 166	    if summary:
 167	        lines.append("")
 168	        lines.append("【概述】（来源: 索引期网关生成 · 已过专名白名单校验）")
 169	        lines.append(summary)
 170	
 171	    mods = card.get("top_modules") or []
 172	    if mods:
 173	        lines.append("")
 174	        lines.append("【顶层模块】（来源: Module 按 loc 降序）")
 175	        for m in mods:
 176	            lines.append(f"- {m.get('path', '?')} (loc={m.get('loc', '?')}): "
 177	                         f"{m.get('doc') or '（无 docstring）'}")
 178	
 179	    hot = card.get("hot_functions") or []
 180	    if hot:
 181	        lines.append("")
 182	        lines.append("【热点函数】（来源: MODIFIES 计数，改动最频繁）")
 183	        for h in hot:
 184	            lines.append(f"- {h.get('qualname', '')} (被 {h.get('modifies', 0)} 次提交修改)")
 185	
 186	    core = card.get("core_concepts") or []
 187	    if core:
 188	        lines.append("")
 189	        lines.append("【核心概念】（来源: IMPLEMENTS 落点最多）")
 190	        for c in core:
 191	            lines.append(f"- {c.get('name', '')} (落 {c.get('impl', 0)} 处实现): "
 192	                         f"{c.get('desc') or ''}")
 193	
 194	    eps = card.get("entrypoints") or []
 195	    lines.append("")
 196	    if eps:
 197	        lines.append("【入口点】（来源: Function.is_endpoint 聚合）")
 198	        for e in eps:
 199	            hm = e.get("http_method") or "?"
 200	            rp = e.get("route_path") or "?"
 201	            lines.append(f"- {e.get('qualname', '')} [{hm} {rp}]")
 202	    else:
 203	        lines.append("【入口点】（来源: Function.is_endpoint 聚合）: （本图无 HTTP 端点）")
 204	
 205	    return "\n".join(lines) + "\n"
