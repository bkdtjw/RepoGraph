  39	def normalize(text: str) -> str:
  40	    """S0 规范化：全半角统一 + 保留标识符大小写 + 保留反引号（供代码词元检测）。
  41	
  42	    - 全角 ASCII 字符（含全角数字/字母/标点 ？，（）等）统一为半角，消除
  43	      "ＡＰＩ" 与 "API"、"？" 与 "?" 的表层分裂；
  44	    - **不做大小写折叠**（标识符大小写载有信息，如 camelCase）；
  45	    - 反引号原样保留——``is_code_token`` 据此把反引号内内容标为强代码词元。
  46	
  47	    纯字符级映射，不改中文内容，不动 CJK 标点 。、（U+3002/U+3001，不在全角 ASCII 段）。
  48	    """
  49	    if not text:
  50	        return ""
  51	    out: list[str] = []
  52	    for ch in text:
  53	        o = ord(ch)
  54	        if ch == _FULLWIDTH_SPACE:
  55	            out.append(" ")
  56	        elif 0xFF01 <= o <= 0xFF5E:
  57	            out.append(chr(o - _FULLWIDTH_OFFSET))
  58	        else:
  59	            out.append(ch)
  60	    return "".join(out)

 170	def _signals(question: str, linked: list, topic_hits: list,

 177	    """

 179	    topic_all_low = not topic_hits
 180	    has_pronoun = bool(_RE_PRONOUN.search(question or ""))
 181	    has_backtick = "`" in (question or "")
 182	    return {
 183	        "has_code_token": bool(has_code_token),
 184	        "no_code_token": not has_code_token,
 185	        "no_linker_hit": no_linker_hit,
 186	        "no_repo_reference": (

 188	        ),
 189	    }

 192	def route(question: str, linked: list, topic_hits: list,
 193	          has_code_token: bool) -> tuple[str, str | None]:
 194	    """规则表按序匹配，返回 ``(label, rule_id)``。
 195	
 196	    参数（均为**链接后**信号，见模块 docstring 的时序）：
 197	    - ``question``       规范化后的问题（``normalize`` 产出）；
 198	    - ``linked``         ``link_entities`` 结果（空 ⇒ ``no_linker_hit``）；
 199	    - ``topic_hits``     ``topic_recall`` 结果（空 ⇒ topic 全低分，供 oos 组合谓词）；
 200	    - ``has_code_token`` ``is_code_token`` 命中。
 201	
 202	    规则命中判定 = 其 ``requires`` 信号全为真 **且**（无 ``pattern`` 或 ``pattern`` 命中）。
 203	    全不中 → ``(entity_local, None)``（``rule_id=None`` 标识兜底，交现有瀑布）。
 204	    """
 205	    sig = _signals(question, linked, topic_hits, has_code_token)
 206	    q = question or ""
 207	    for rule in ROUTER_RULES:
 208	        reqs = rule.get("requires") or ()
 209	        if not all(sig.get(name, False) for name in reqs):
 210	            continue
 211	        pat = rule.get("pattern")
 212	        if pat is not None and not pat.search(q):
 213	            continue
 214	        return rule["label"], rule["id"]
 215	    return _FALLBACK_LABEL, None
