 589	def _rg_cache_put(cache, key, val, cap=_RG_REWRITE_CACHE_MAX):
 590	    """LRU-ish 写缓存 (dict 保序, 满则逐出最早)。"""
 591	    if key in cache:
 592	        cache.pop(key)
 593	    elif len(cache) >= cap:
 594	        cache.pop(next(iter(cache)))
 595	    cache[key] = val

 599	    """从模型返回文本解析 {queries, symbol_guesses, premises} (容错; 失败返回空结构)。"""
 600	    empty = {"queries": [], "symbol_guesses": [], "premises": []}

 620	def _rg_llm_rewrite(cfg, question, recent_turns=None):
 621	    """S2: 一次 flash 调用出 {queries, symbol_guesses, premises} (附录 B; 走 _rg_llm_json 底座)。
 622	
 623	    - 按问题做 LRU 缓存 (规范化前的原问题作键即可, 同问免重复往返);
 624	    - ``recent_turns`` 取最近数轮做指代消解上下文 (可选);
 625	    - 网关缺配置 / 失败 / 解析失败 → 空结构 (静默降级, 绝不抛、绝不回显 token)。
 626	    """
 627	    q = (question or "").strip()
 628	    if not q:
 629	        return {"queries": [], "symbol_guesses": [], "premises": []}
 630	    cached = _RG_REWRITE_CACHE.get(q)
 631	    if cached is not None:
 632	        return cached
 633	    ctx_hint = ""
 634	    if recent_turns:
 635	        joined = " / ".join(str(t)[:120] for t in list(recent_turns)[-3:] if t)
 636	        if joined:
 637	            ctx_hint = "\n【最近对话（供指代消解）】" + joined
 638	    user = "问题：" + q + ctx_hint + "\n\n只输出 JSON："
 639	    text = _rg_llm_json(cfg, _RG_REWRITE_SYSTEM, user, timeout=20, max_tokens=400)
 640	    result = _parse_rewrite(text)
 641	    _rg_cache_put(_RG_REWRITE_CACHE, q, result)
 642	    return result

 645	def _rg_try_rewrite_link(cfg, question, recent_turns=None):
 646	    """S2 改写 → 二次链接/主题召回 (extra_queries 传入 build_repo_context)。
 647	
 648	    entity_local 无主锚 (mode∈overview/none) 时的主召回补偿 (D-N2)。返回
 649	    {mode,linked,stats,context_text,resolved_query,symbol_guesses,premises} 或 None (未改善)。
 650	    绝不抛异常; build_ctx 不支持 extra_queries (旧签名) → None (静默不启用)。
 651	    """
 652	    store = _RG.get("store")
 653	    build_ctx = _RG.get("build_ctx")
 654	    if store is None or build_ctx is None:
 655	        return None
 656	    if "extra_queries" not in _fn_param_names(build_ctx):
 657	        return None
 658	    try:
 659	        rw = _rg_llm_rewrite(cfg, question, recent_turns=recent_turns)
 660	        queries = rw.get("queries") or []
 661	        # symbol_guesses 也并入 extra_queries (只作检索辅助, 永不进答案事实)
 662	        extra = list(queries) + list(rw.get("symbol_guesses") or [])
 663	        if not extra:
 664	            return None
 665	        corpus_index = _RG.get("corpus")
 666	        res = _rg_call_build_ctx(build_ctx, store, question, corpus_index, True,
 667	                                 extra_queries=extra)
 668	        if not isinstance(res, dict):
 669	            return None
 670	        mode = _rg_normalize_mode(res.get("mode"))
 671	        ctx = res.get("context_text") or ""
 672	        # 仅在二次召回确实落到 symbol/topic/llm 上下文时采信 (overview/none = 仍无锚, 交 L2)
 673	        if mode not in ("symbol", "topic", "llm") or not ctx:
 674	            return None

 676	                "stats": res.get("stats") or {}, "context_text": ctx,
 677	                "resolved_query": " / ".join(queries[:2]) if queries else "",
 678	                "symbol_guesses": rw.get("symbol_guesses") or [],
 679	                "premises": rw.get("premises") or []}

 696	def _rg_collect_premise_flags(cfg, content, rg, semantic_mode, recent_turns):
 697	    """合并 S7 前提 flag: lexical (build_repo_context) + (overview/none·llm) 校验后的 LLM premises。
 698	
 699	    真实校验: LLM premises 经 router.verify_premises 对真实 store 做图边存在性校验, 绝不臆造;
 700	    网关缺失/异常静默降级为仅 lexical。返回去重后的 flag 列表 [{claim,status,...}]。
 701	    """
 702	    flags = []
 703	    seen = set()
 704	    for f in (rg.get("premise_flags") or []):          # lexical (离线规则, 已校验)
 705	        if isinstance(f, dict):
 706	            claim = f.get("claim")
 707	            fl = f
 708	        else:
 709	            claim, fl = str(f), {"claim": str(f), "status": "unverified"}
 710	        if claim and claim not in seen:
 711	            seen.add(claim)
 712	            flags.append(fl)
 713	    # llm 档补充 (仅 overview/none 且 build_ctx 支持 extra_queries —— rewrite 已搭车缓存)
 714	    build_ctx = _RG.get("build_ctx")
 715	    store = _RG.get("store")
 716	    if (semantic_mode == "llm" and gateway_base(cfg) and store is not None
 717	            and build_ctx is not None
 718	            and "extra_queries" in _fn_param_names(build_ctx)):
 719	        try:
 720	            from repograph.retrieve.router import verify_premises, premises_from_claims
 721	            rw = _rg_llm_rewrite(cfg, content, recent_turns=recent_turns)   # 缓存命中
 722	            claims = rw.get("premises") or []
 723	            for f in verify_premises(store, premises_from_claims(claims)):
 724	                claim = f.get("claim")
 725	                if claim and claim not in seen:
 726	                    seen.add(claim)
 727	                    flags.append(f)
 728	        except Exception:  # noqa: BLE001 —— 前提校验失败静默降级为仅 lexical, 绝不打断聊天
 729	            pass
 730	    return flags

 733	def _rg_premise_gate_text(flags):
 734	    """构造前提闸门注入前缀 (spec §5.2)。非空 flag 时要求生成层先纠正后答, 不顺预设作答。"""
 735	    if not flags:
 736	        return ""
 737	    lines = [
 738	        "【前提校验闸门】(来源: RepoGraph 图边存在性校验, S7)",
 739	        "以下前提未获代码知识图谱证据, 回答前必须先明确纠正、不得顺着这些预设作答:",
 740	    ]
 741	    for f in flags:
 742	        claim = (f.get("claim") if isinstance(f, dict) else str(f)) or ""
 743	        lines.append(
 744	            f"⚠ 前提未获图谱证据: {claim}; 据图谱实际: 本仓库图谱中查无「{claim}」的相关实体或边, "
 745	            f"该技术/结构很可能并未在本项目中使用, 请先纠正此前提再回答。")
 746	    return "\n".join(lines) + "\n\n"
