 215	def _summary_input(card: dict, readme_head: str = "") -> tuple[str, set[str]]:
 216	    """构造 summary 生成的输入文本 + 允许出现的专名白名单集合（英文标识符/路径）。"""
 217	    parts: list[str] = [render_card_text({**card, "summary": None})]
 218	    if readme_head:
 219	        parts.append("README 首段：" + readme_head)
 220	    input_text = "\n".join(parts)
 221	    allowed = {m.group(0) for m in _IDENT_RE.finditer(input_text)}
 222	    return input_text, allowed

 225	def _summary_violates_whitelist(summary: str, input_text: str) -> list[str]:
 226	    """返回 summary 里**不在输入中**的英文标识符（专名白名单违规项，反幻觉 F2）。
 227	
 228	    只校验英文标识符类专名（发明 Redis/FastAPI 之类的主要幻觉风险面）；短词（≤1 字符）
 229	    与在输入文本中出现（子串）者放行。中文由生成侧 prompt 约束 + 人工抽检兜底。
 230	    """
 231	    low_input = input_text.lower()
 232	    bad: list[str] = []
 233	    for m in _IDENT_RE.finditer(summary):
 234	        tok = m.group(0)
 235	        if len(tok) <= 1:
 236	            continue
 237	        if tok.lower() not in low_input:
 238	            bad.append(tok)
 239	    return bad

 242	def generate_card_summary(store: GraphStore, card: Optional[dict] = None,
 243	                          readme_head: str = "", model: Optional[str] = None,
 244	                          config_path: Optional[str] = None,
 245	                          whitelist_retries: int = 4) -> Optional[str]:
 246	    """唯一一次索引期真实网关调用生成 ≤300 字 summary；白名单违规重试后降级弃之。
 247	
 248	    真实数据铁律：网络/网关**真实发生**；任何失败（配置缺失 / HTTP 错误 / 白名单违规
 249	    达 ``whitelist_retries`` 次）→ 返回 ``None``（弃 summary），**绝不伪造**。绝不回显 token
 250	    （由 llm_client 保证）。白名单违规靠重试而非放宽校验消化（模型输出非确定性，反幻觉防线不降）。
 251	    """
 252	    from ..extract.llm_client import ask_gateway, INDEX_MODEL, GatewayConfigError, GatewayCallError
 253	
 254	    if card is None:
 255	        card = build_repo_card(store)
 256	    input_text, _allowed = _summary_input(card, readme_head)
 257	    system = (
 258	        "你是代码库概述撰写器。根据下面提供的【确定性事实】（规模统计、顶层模块、热点函数、"
 259	        "核心概念）为该代码库写一段 ≤300 字的中文概述，供他人快速了解这个项目是做什么的、"
 260	        "整体结构与重点。严格约束：(1) 只能使用事实中原样出现过的专名（模块路径、函数名、概念名、"
 261	        "数字），严禁发明事实里没有的技术栈名词、库名或英文缩写；(2) 尽量用中文表述，不自造英文词；"
 262	        "(3) 不做评价性夸张；(4) 只输出概述正文，不要标题、不要 JSON、不要解释。"
 263	    )
 264	    user = "【确定性事实】\n" + input_text + "\n\n请据此写 ≤300 字中文概述："
 265	
 266	    for _attempt in range(max(1, whitelist_retries)):
 267	        try:
 268	            # qwen3.8-max-preview 为大模型、时延波动大，给足 timeout（90s）；ask_gateway
 269	            # 内部再重试 1 次消化瞬时超时，外层 whitelist_retries 消化非确定性白名单违规。
 270	            text = ask_gateway(system, user, model=model or INDEX_MODEL,
 271	                               max_tokens=600, config_path=config_path,
 272	                               timeout=90, retries=1)
 273	        except (GatewayConfigError, GatewayCallError):
 274	            return None                      # 如实降级，不伪造
 275	        summary = " ".join(text.split()).strip()
 276	        if len(summary) > _SUMMARY_MAX_CHARS:
 277	            summary = summary[:_SUMMARY_MAX_CHARS]
 278	        bad = _summary_violates_whitelist(summary, input_text)
 279	        if not bad:
 280	            return summary
 281	        # 白名单违规（模型输出非确定性）：重试；耗尽仍违规则弃 summary（降级）
 282	    return None

 285	def generate_and_save(store: GraphStore, cache_path: Optional[str] = None,
 286	                      readme_head: str = "", model: Optional[str] = None,
 287	                      config_path: Optional[str] = None) -> dict:
 288	    """索引期一次性：确定性卡片 + 真实 summary → 写 ``output/repo_card.json``。
 289	
 290	    summary 生成失败时只落确定性字段（``summary=null``），meta 路由仍可作答（与
 291	    ``cli.py`` 既有「语义层未就绪降级为告警」一致）。返回落盘的卡片 dict。
 292	    """
 293	    path = cache_path or default_cache_path()
 294	    card = build_repo_card(store)
 295	    summary = generate_card_summary(store, card, readme_head=readme_head,
 296	                                    model=model, config_path=config_path)
 297	    card["summary"] = summary
 298	    os.makedirs(os.path.dirname(path) or ".", exist_ok=True)
 299	    with open(path, "w", encoding="utf-8") as f:
 300	        json.dump(card, f, ensure_ascii=False, indent=1)
 301	    return card

 308	def load_or_build_repo_card(store: GraphStore,
 309	                            cache_path: Optional[str] = None) -> tuple[dict, bool]:
 310	    """读取 ``output/repo_card.json`` 缓存；缺失/损坏 → 现场 ``build_repo_card`` + ``degraded=True``。
 311	
 312	    返回 ``(card, degraded)``。缓存损坏/结构不符时不抛错（P4 绝不因缺文件裸拒），只降级。
 313	    """
 314	    path = cache_path or default_cache_path()
 315	    if os.path.exists(path):
 316	        try:
 317	            with open(path, "r", encoding="utf-8") as f:
 318	                card = json.load(f)
 319	            if isinstance(card, dict) and isinstance(card.get("stats"), dict):
 320	                return card, False
 321	        except Exception:  # noqa: BLE001  损坏缓存 → 现场重建
 322	            pass
 323	    return build_repo_card(store), True

 327	    """meta 路由：注入 repo_card（缓存优先），返回 ``mode='meta'`` 的上下文 dict。
 328	
 329	    返回统一 ``{mode, linked, context_text, stats, degraded}``；``degraded`` 标识本次是否
 330	    走了现场降级（缓存缺失/损坏）。绝不裸拒（context_text 恒含规模事实）。
 331	    """
 332	    card, degraded = load_or_build_repo_card(store, cache_path)
 333	    s = card.get("stats", {})
 334	    stats = {
 335	        "symbols": 0, "topics": 0, "impact_callers": 0, "commits": 0,
 336	        "concepts": s.get("concepts", 0),
 337	        "modules": s.get("modules", 0), "functions": s.get("functions", 0),
 338	        "classes": s.get("classes", 0),
 339	    }
 340	    # mode 取 overview 展示态（spec §5.1：build_repo_context 侧恒返回 overview 类展示 mode，

 342	    # 精确五分类，mode='overview' 保证 server._RG_VALID_MODES 兼容、零 server 改动）。
 343	    return {
