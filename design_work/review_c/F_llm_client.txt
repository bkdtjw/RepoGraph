  95	def _extract_text(raw: bytes) -> str:
  96	    """从非流式 messages 响应体提取拼接文本块（与 server._extract_text 同义）。"""
  97	    try:
  98	        obj = json.loads(raw.decode("utf-8", "replace"))
  99	    except Exception:  # noqa: BLE001
 100	        return ""
 101	    if isinstance(obj, dict):
 102	        content = obj.get("content")
 103	        if isinstance(content, list):
 104	            parts = [(b.get("text") or "") for b in content
 105	                     if isinstance(b, dict) and b.get("type") == "text"]
 106	            if parts:
 107	                return "".join(parts)
 108	    return ""

 115	def ask_gateway(system: str, user: str, model: str | None = None,
 116	                max_tokens: int = 512, timeout: int = 30,
 117	                config_path: str | None = None,
 118	                retries: int = 2, sleep_s: float = 0.5) -> str:
 119	    """向网关发一次**非流式** messages 请求，返回首个文本块（拼接）。
 120	
 121	    - ``model`` 缺省用 ``INDEX_MODEL``（qwen3.8-max-preview）；
 122	    - 鉴权令牌只入 ``Authorization`` 头，**绝不回显**；异常消息不含 token；
 123	    - 限速（``sleep_s``）+ 重试（``retries`` 次）应对瞬时失败；
 124	    - 全部重试失败 → 抛 ``GatewayCallError``（调用方按需降级，**不伪造输出**）。
 125	
 126	    配置缺失（无文件 / 无 base / 无 token）→ 抛 ``GatewayConfigError``。
 127	    """
 128	    cfg = load_gateway_config(config_path)
 129	    base, token = _resolve_endpoint(cfg)
 130	    mdl = model or INDEX_MODEL
 131	    url = base + "/v1/messages"
 132	    payload = {
 133	        "model": mdl,
 134	        "max_tokens": max_tokens,
 135	        "system": system,
 136	        "messages": [{"role": "user", "content": user}],
 137	    }
 138	    data = json.dumps(payload).encode("utf-8")
 139	    headers = {
 140	        "Authorization": "Bearer " + token,        # 仅入请求头，绝不打印/记录
 141	        "anthropic-version": ANTHROPIC_VERSION,
 142	        "content-type": "application/json",
 143	    }
 144	
 145	    last_err = "unknown"
 146	    attempts = max(1, retries + 1)
 147	    for i in range(attempts):
 148	        req = urllib.request.Request(url, data=data, headers=headers, method="POST")
 149	        try:
 150	            with urllib.request.urlopen(req, timeout=timeout) as resp:
 151	                raw = resp.read()
 152	            text = _extract_text(raw)
 153	            if text.strip():
 154	                return text
 155	            last_err = "响应无文本块"
 156	        except urllib.error.HTTPError as e:      # 不读 e 的鉴权相关字段，不外泄
 157	            last_err = f"HTTP {e.code}"
 158	        except Exception as e:                   # noqa: BLE001  超时 / 网络 / 解析
 159	            last_err = type(e).__name__
 160	        if i < attempts - 1:
 161	            time.sleep(sleep_s)
 162	    raise GatewayCallError(f"网关调用失败（{mdl}）：{last_err}")
