 475	def premises_from_claims(claims: list) -> list[dict]:
 476	    """把 LLM 抽取的前提断言（自然语言字符串）转成可校验 premise dict。
 477	
 478	    从每条断言抽取**技术专名 + 英文标识符 + 数字量词短语**作为 terms（供图谱存在性校验）。
 479	    抽不出可校验 term 的纯中文泛述断言**不产 premise**（保守：宁可漏判也不误报，守 F1）。
 480	    """
 481	    from .lexicon import find_tech_terms
 482	    out: list[dict] = []
 483	    for c in claims or []:
 484	        s = str(c or "").strip()
 485	        if not s:
 486	            continue
 487	        terms: list[str] = [t for t, _d in find_tech_terms(s)]
 488	        terms += _EN_IDENT_RE.findall(s)
 489	        terms += [re.sub(r"\s+", "", m) for m in _NUM_UNIT_RE.findall(s)]
 490	        # 去重保序 + 去掉过短碎片
 491	        terms = list(dict.fromkeys(t for t in terms if t and len(t) >= 2))
 492	        if not terms:
 493	            continue
 494	        out.append({"claim": s, "terms": terms, "source": "llm:premise"})
 495	    return out

 498	def _term_in_graph(store, term: str) -> bool:
 499	    """term（忽略大小写）是否作为词面出现在图谱任一节点的可检索文本中。
 500	
 501	    扫 ``_PREMISE_SCAN_KEYS`` + aliases/zh_aliases + evidence 引文。命中即认为该前提实体
 502	    在图谱「可定位」（等价 link_entities 的词面可链接性，但覆盖概念描述/提交信息等自由文本）。
 503	    """
 504	    t = (term or "").lower()
 505	    if not t:
 506	        return False
 507	    for node in store.nodes():
 508	        for k in _PREMISE_SCAN_KEYS:
 509	            v = node.get(k)
 510	            if v and t in str(v).lower():
 511	                return True
 512	        for a in list(node.get("aliases") or []) + list(node.get("zh_aliases") or []):
 513	            if a and t in str(a).lower():
 514	                return True
 515	        for ev in node.get("evidence") or []:
 516	            q = (ev or {}).get("quote") if isinstance(ev, dict) else None
 517	            if q and t in str(q).lower():
 518	                return True
 519	    return False

 522	def verify_premises(store, premises: list) -> list[dict]:
 523	    """S7 前提校验：逐前提对真实图谱做词面存在性校验，返回**未证实**的 flag 列表。
 524	
 525	    每个 premise 形如 ``{claim, terms:[...], source}``。判定（落地设计 §5.7 / D-19）：
 526	      · terms **全部**不可词面定位于图谱 → ``status='unverified'`` + ``reason='unknown_entity'``
 527	        （实体不可链接，§3.2 S7 保守当作未证实）；
 528	      · terms 有可定位者 → 本轮词面校验保守视为「已获证据」，**不产 flag**（可定位但断言
 529	        支撑边缺失的 `missing_edge` 半支需边级校验，留待具体关系断言，避免误伤真前提）。
 530	
 531	    返回 ``[{claim, status:'unverified', reason, source, terms}]``（仅未证实项；已证实前提不产 flag）。
 532	    真实数据铁律：只据真实 ``store`` 扫描判定，绝不臆造。
 533	    """
 534	    flags: list[dict] = []
 535	    for prem in premises or []:
 536	        terms = [t for t in (prem.get("terms") or []) if t]
 537	        if not terms:
 538	            continue
 539	        present = any(_term_in_graph(store, t) for t in terms)
 540	        if present:
 541	            continue                        # 可定位 → 保守不产 flag（不误伤真前提）
 542	        flags.append({
 543	            "claim": prem.get("claim", ""),
 544	            "status": "unverified",
 545	            "reason": "unknown_entity",
 546	            "source": prem.get("source"),
 547	            "terms": terms,
 548	        })
 549	    return flags
