 337| def call_api(system: str | None, user: str, max_tokens: int, retries: int) -> dict:
 338|     """POST {base}/v1/messages（Anthropic 格式）。返回 {ok, text, usage, stop_reason, attempts, error}。
 339| 
 340|     重试 retries 次（共 retries+1 次尝试）；每次尝试后 sleep(SLEEP) 限速。
 341|     仅记录可写盘字段；api_key/host 不出现在返回值里。
 342|     """
 343|     cfg = _cfg()
 344|     base = cfg["base_url"].rstrip("/")
 345|     key = cfg["api_key"]
 346|     model = cfg["model"]
 347|     ah = cfg["auth_header"]
 348|     body = {"model": model, "max_tokens": max_tokens,
 349|             "messages": [{"role": "user", "content": user}]}
 350|     if system:
 351|         body["system"] = system
 352|     raw = json.dumps(body).encode("utf-8")
 353| 
 354|     last_err = ""
 355|     attempts = 0
 356|     for attempt in range(retries + 1):
 357|         attempts = attempt + 1
 358|         try:
 359|             req = urllib.request.Request(base + "/v1/messages", data=raw, method="POST")
 360|             req.add_header("content-type", "application/json")
 361|             req.add_header(ah, key)
 362|             req.add_header("anthropic-version", "2023-06-01")
 363|             with urllib.request.urlopen(req, timeout=HTTP_TIMEOUT) as r:
 364|                 data = json.loads(r.read().decode("utf-8"))
 365|             text = "".join(b.get("text", "") for b in data.get("content", [])
 366|                            if isinstance(b, dict) and b.get("type") == "text")
 367|             if not text:
 368|                 text = "".join(b.get("text", "") for b in data.get("content", [])
 369|                                if isinstance(b, dict))
 370|             time.sleep(SLEEP)
 371|             return {"ok": True, "text": text, "usage": data.get("usage"),
 372|                     "stop_reason": data.get("stop_reason"),
 373|                     "resp_model": data.get("model"), "attempts": attempts, "error": ""}
 374|         except urllib.error.HTTPError as e:
 375|             try:
 376|                 detail = e.read().decode("utf-8", "replace")[:300]
 377|             except Exception:
 378|                 detail = ""
 379|             last_err = f"HTTP {e.code} {e.reason} {detail}"
 380|         except Exception as e:
 381|             last_err = f"{type(e).__name__}: {str(e)[:200]}"
 382|         time.sleep(SLEEP)
 383|     return {"ok": False, "text": "", "usage": None, "stop_reason": None,
 384|             "resp_model": None, "attempts": attempts, "error": last_err}

 516| def parse_verdict(text: str) -> tuple:
 517|     """从裁判响应抽 {verdict, reason}；容忍 ```json 包裹与前后噪声。失败返回 (None, None)。"""
 518|     if not text:
 519|         return None, None
 520|     s = text.strip()
 521|     if "```" in s:
 522|         # 取首个代码块内内容
 523|         parts = s.split("```")
 524|         for seg in parts:
 525|             seg2 = seg.strip()
 526|             if seg2.startswith("json"):
 527|                 seg2 = seg2[4:].strip()
 528|             if seg2.startswith("{"):
 529|                 s = seg2
 530|                 break
 531|     i, j = s.find("{"), s.rfind("}")
 532|     if i < 0 or j < 0 or j <= i:
 533|         return None, None
 534|     try:
 535|         obj = json.loads(s[i:j + 1])
 536|     except Exception:
 537|         return None, None
 538|     v = str(obj.get("verdict", "")).strip().lower()
 539|     if v not in ("correct", "partial", "wrong"):
 540|         return None, None
 541|     return v, str(obj.get("reason", ""))[:80]
